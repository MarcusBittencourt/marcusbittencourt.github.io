Plot radar graph

library(plotly)

p <- plot_ly(
    type = 'scatterpolar',
    r = c(39, 28, 8, 7, 28, 39),
    theta = c('A','B','C', 'D', 'E', 'A'),
    fill = 'toself'
) %>%
layout(
    polar = list(
    radialaxis = list(
        visible = T,
        range = c(0,50)
    )
    ),
    showlegend = F
)
